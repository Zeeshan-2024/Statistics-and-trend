def plot_histogram():
    plt.figure(figsize=(8, 6))
    plt.hist(dataset['Age'], bins=20, color='skyblue')
    plt.title('Age Distribution')
    plt.xlabel('Age')
    plt.ylabel('Frequency')
    plt.show()
